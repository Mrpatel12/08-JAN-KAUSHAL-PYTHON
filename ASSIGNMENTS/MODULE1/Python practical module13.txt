    count = 0
    num = 0
    while count < 10:
        yield num
        num += 2
        count += 1

2). Write a Python program that uses a custom iterator to iterate over a list of integers.

Ans.

class CustomIterator:
    def __init__(self, data):
        self.data = data
        self.index = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        current_item = self.data[self.index]
        self.index += 1
        return current_item

my_list = [10, 20, 30, 40, 50]
custom_iterator = CustomIterator(my_list)

print("Iterating over the list using the custom iterator:")
for number in custom_iterator:
    print(number)

print("\nDemonstrating manual iteration:")
iterator_manual = iter(my_list) # Use the built-in iter for comparison

try:
    print(next(iterator_manual))
    print(next(iterator_manual))
    iterator_custom_manual = CustomIterator(my_list)
    print(next(iterator_custom_manual))
except StopIteration:
    print("Iteration finished.")

2).Write a Python program that uses reduce() to find the product of a list of numbers

Ans.
from functools import reduce

def find_product(numbers):
 
  Args:
    numbers: A list of integers or floats.

  Returns:
    The product of all numbers in the list.

  product = reduce(lambda x, y: x * y, numbers)
  return product

my_list = [2, 3, 5, 4]
result = find_product(my_list)
print(f"The list is: {my_list}")
print(f"The product of the list using reduce() is: {result}")


list_floats = [1.5, 2.0, 3.0]
result_floats = find_product(list_floats)
print(f"\nThe list is: {list_floats}")
print(f"The product of the list using reduce() is: {result_floats}")
